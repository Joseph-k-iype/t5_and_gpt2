def merge_process_cells_in_table(table, df):
    """
    Merge cells in the first column ("Process") for identical process names in the PowerPoint table.
    Parameters:
    - table: The PowerPoint table object.
    - df: The DataFrame used to populate the table.
    """
    process_column_idx = 0  # Index of the "Process" column in the table
    merge_start = None  # Start row for merging
    previous_value = None

    for row_idx, row in enumerate(df.itertuples(index=False), start=1):  # Start at row 1 to skip headers
        current_value = row.Process

        if current_value == previous_value:
            # Continue the merge group
            if merge_start is None:
                merge_start = row_idx - 1  # Mark the start of the merge
        else:
            # Finalize the previous merge
            if merge_start is not None:
                table.cell(merge_start, process_column_idx).merge(table.cell(row_idx - 1, process_column_idx))
                merge_start = None  # Reset the merge start

        previous_value = current_value

    # Finalize the last merge group
    if merge_start is not None:
        table.cell(merge_start, process_column_idx).merge(table.cell(len(df), process_column_idx) - 1)
