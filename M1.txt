def merge_process_cells_in_table(table, df):
    """
    Merge cells in the first column ("Process") for identical process names in the PowerPoint table.
    Parameters:
    - table: The PowerPoint table object.
    - df: The DataFrame used to populate the table.
    """
    process_column_idx = 0  # Index of the "Process" column in the table
    rows_to_merge = []
    previous_value = None

    # Iterate through rows in the DataFrame
    for row_idx, row in enumerate(df.itertuples(index=False), start=1):  # Start from row 1 (data rows)
        current_value = row.Process  # Access the "Process" column value

        if current_value == previous_value:
            # Add the row to the current merge group
            rows_to_merge[-1].append(row_idx)
        else:
            # Start a new merge group
            rows_to_merge.append([row_idx])

        previous_value = current_value

    # Perform merging for each group
    for merge_group in rows_to_merge:
        if len(merge_group) > 1:
            try:
                # Ensure the indices are within the table row range
                if merge_group[-1] < len(table.rows):
                    table.cell(merge_group[0], process_column_idx).merge(
                        table.cell(merge_group[-1], process_column_idx)
                    )
            except IndexError as e:
                print(f"Error merging rows {merge_group}: {e}")
