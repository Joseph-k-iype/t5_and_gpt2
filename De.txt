import React, { useState } from 'react';
import { BarChart3, TrendingUp, Database, Zap } from 'lucide-react';
import LoadingSpinner from './LoadingSpinner';
import api from '../services/api';

const StatisticsPanel = () => {
  const [statsData, setStatsData] = useState(null);
  const [loading, setLoading] = useState(false);

  const fetchStats = async () => {
    setLoading(true);
    const { data } = await api.getStats();
    if (data) {
      setStatsData(data);
    }
    setLoading(false);
  };

  const StatCard = ({ title, items, icon: Icon, color }) => (
    <div className="card p-6">
      <div className="flex items-center gap-3 mb-4">
        <div className={`p-2 rounded-lg ${color.bg}`}>
          <Icon className={`w-5 h-5 ${color.text}`} />
        </div>
        <h3 className="text-lg font-semibold text-gray-900">{title}</h3>
      </div>
      <div className="space-y-3">
        {items.map((item, index) => (
          <div
            key={index}
            className="flex items-center justify-between p-3 bg-gray-50 rounded-lg"
          >
            <span className="text-sm text-gray-600">{item.label}</span>
            <span className="text-lg font-bold text-gray-900">{item.value}</span>
          </div>
        ))}
      </div>
    </div>
  );

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <h2 className="text-xl font-semibold text-gray-900">Detailed Statistics</h2>
        <button
          onClick={fetchStats}
          disabled={loading}
          className="btn-primary flex items-center gap-2"
        >
          {loading ? (
            <>
              <LoadingSpinner size="sm" />
              <span>Loading...</span>
            </>
          ) : (
            <>
              <BarChart3 className="w-4 h-4" />
              <span>Load Statistics</span>
            </>
          )}
        </button>
      </div>

      {!statsData && !loading && (
        <div className="card p-12 text-center">
          <BarChart3 className="w-16 h-16 text-gray-300 mx-auto mb-4" />
          <h3 className="text-lg font-medium text-gray-900 mb-2">
            No Statistics Loaded
          </h3>
          <p className="text-gray-500 mb-4">
            Click the "Load Statistics" button to view detailed metrics
          </p>
        </div>
      )}

      {loading && !statsData && (
        <div className="flex items-center justify-center py-12">
          <LoadingSpinner size="lg" />
        </div>
      )}

      {statsData && (
        <div className="space-y-6">
          {/* Total Unique Cases */}
          <StatCard
            title="Total Unique Cases"
            icon={Database}
            color={{ bg: 'bg-blue-50', text: 'text-blue-600' }}
            items={[
              {
                label: 'Application IDs',
                value: statsData.total_unique_cases.application_ids.toLocaleString(),
              },
              {
                label: 'DV Case IDs',
                value: statsData.total_unique_cases.dv_case_ids.toLocaleString(),
              },
              {
                label: 'Business Application IDs',
                value: statsData.total_unique_cases.business_ids.toLocaleString(),
              },
            ]}
          />

          {/* Compliance Status */}
          <StatCard
            title="Compliance Status"
            icon={TrendingUp}
            color={{ bg: 'bg-emerald-50', text: 'text-emerald-600' }}
            items={[
              {
                label: 'PIA Completed',
                value: statsData.compliance_status.pia_completed.toLocaleString(),
              },
              {
                label: 'TIA Completed',
                value: statsData.compliance_status.tia_completed.toLocaleString(),
              },
            ]}
          />

          {/* Performance Metrics */}
          <div className="card p-6">
            <div className="flex items-center gap-3 mb-4">
              <div className="p-2 rounded-lg bg-purple-50">
                <Zap className="w-5 h-5 text-purple-600" />
              </div>
              <h3 className="text-lg font-semibold text-gray-900">
                Performance Metrics
              </h3>
            </div>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <div className="p-4 bg-gradient-to-br from-purple-50 to-purple-100 rounded-lg border border-purple-200">
                <p className="text-sm text-purple-700 mb-1">Lookup Complexity</p>
                <p className="text-2xl font-bold text-purple-900">
                  {statsData.performance.lookup_complexity}
                </p>
              </div>
              <div className="p-4 bg-gradient-to-br from-purple-50 to-purple-100 rounded-lg border border-purple-200">
                <p className="text-sm text-purple-700 mb-1">Module Check Complexity</p>
                <p className="text-2xl font-bold text-purple-900">
                  {statsData.performance.module_check_complexity}
                </p>
              </div>
            </div>
          </div>

          {/* Summary Card */}
          <div className="card p-6 bg-gradient-to-r from-primary-50 to-accent-50 border-primary-200">
            <h3 className="text-lg font-semibold text-gray-900 mb-3">Summary</h3>
            <div className="space-y-2 text-sm text-gray-700">
              <p>
                • Total tracked cases:{' '}
                <span className="font-semibold">
                  {(
                    statsData.total_unique_cases.application_ids +
                    statsData.total_unique_cases.dv_case_ids +
                    statsData.total_unique_cases.business_ids
                  ).toLocaleString()}
                </span>
              </p>
              <p>
                • Total completed assessments:{' '}
                <span className="font-semibold">
                  {(
                    statsData.compliance_status.pia_completed +
                    statsData.compliance_status.tia_completed
                  ).toLocaleString()}
                </span>
              </p>
              <p>
                • System performance: <span className="font-semibold">Optimized for O(1) lookups</span>
              </p>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default StatisticsPanel;
