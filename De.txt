    <!-- Inference Logic -->
    <!-- Rule 1: Applications hosted in a GDPR Country -->
    <rdf:Description rdf:about="#Application">
        <rdfs:subClassOf>
            <owl:Restriction>
                <owl:onProperty rdf:resource="#hostedIn"/>
                <owl:someValuesFrom rdf:resource="#Country"/>
            </owl:Restriction>
        </rdfs:subClassOf>
        <rdfs:comment>Applications hosted in GDPR-compliant countries (e.g., Germany) must adhere to GDPR requirements.</rdfs:comment>
    </rdf:Description>

    <!-- Rule 2: Applications hostedOn DataCenters must support Data Lineage -->
    <rdf:Description rdf:about="#DataLineage">
        <rdfs:subClassOf>
            <owl:Restriction>
                <owl:onProperty rdf:resource="#hostedOn"/>
                <owl:someValuesFrom rdf:resource="#DataCenter"/>
            </owl:Restriction>
        </rdfs:subClassOf>
        <rdfs:comment>Data Lineage processes rely on applications hosted in compliant data centers.</rdfs:comment>
    </rdf:Description>

    <!-- Rule 3: Reuse of Data Lineage -->
    <rdf:Description rdf:about="#DataLineage">
        <rdfs:subClassOf>
            <owl:Restriction>
                <owl:onProperty rdf:resource="#supportedBy"/>
                <owl:allValuesFrom rdf:resource="#CustomerData"/>
            </owl:Restriction>
        </rdfs:subClassOf>
        <rdfs:comment>CustomerData lineage for GDPR compliance is reusable for RiskReporting in BCBS 239 and PRA.</rdfs:comment>
    </rdf:Description>

    <!-- Rule 4: Regulatory Scope Based on Geography -->
    <rdf:Description rdf:about="#Regulation">
        <rdfs:subClassOf>

            </owl:Restriction>
        </rdfs:subClassOf>
        <rdfs:comment>Different regulations apply based on the countries in which applications are hosted.</rdfs:comment>
    </rdf:Description>
