import React, { useEffect, useState } from 'react';
import { Activity, RefreshCw, Database, CheckCircle2 } from 'lucide-react';
import LoadingSpinner from './LoadingSpinner';
import api from '../services/api';

const HealthMonitor = () => {
  const [healthData, setHealthData] = useState(null);
  const [loading, setLoading] = useState(true);

  const fetchHealth = async () => {
    setLoading(true);
    const { data } = await api.getHealth();
    if (data) {
      setHealthData(data);
    }
    setLoading(false);
  };

  useEffect(() => {
    fetchHealth();
  }, []);

  if (loading && !healthData) {
    return (
      <div className="flex items-center justify-center py-12">
        <LoadingSpinner size="lg" />
      </div>
    );
  }

  if (!healthData) {
    return (
      <div className="card p-6 text-center">
        <p className="text-gray-500">Unable to fetch health data</p>
      </div>
    );
  }

  const stats = [
    {
      label: 'System Status',
      value: healthData.status,
      icon: Activity,
      color: healthData.status === 'healthy' ? 'text-green-600' : 'text-amber-600',
      bgColor: healthData.status === 'healthy' ? 'bg-green-50' : 'bg-amber-50',
    },
    {
      label: 'Application IDs',
      value: healthData.indexes?.application_ids?.toLocaleString() || '0',
      icon: Database,
      color: 'text-blue-600',
      bgColor: 'bg-blue-50',
    },
    {
      label: 'DV Case IDs',
      value: healthData.indexes?.dv_case_ids?.toLocaleString() || '0',
      icon: Database,
      color: 'text-purple-600',
      bgColor: 'bg-purple-50',
    },
    {
      label: 'Business IDs',
      value: healthData.indexes?.business_ids?.toLocaleString() || '0',
      icon: Database,
      color: 'text-indigo-600',
      bgColor: 'bg-indigo-50',
    },
    {
      label: 'PIA Completed',
      value: healthData.indexes?.pia_completed_cases?.toLocaleString() || '0',
      icon: CheckCircle2,
      color: 'text-emerald-600',
      bgColor: 'bg-emerald-50',
    },
    {
      label: 'TIA Completed',
      value: healthData.indexes?.tia_completed_cases?.toLocaleString() || '0',
      icon: CheckCircle2,
      color: 'text-teal-600',
      bgColor: 'bg-teal-50',
    },
  ];

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <h2 className="text-xl font-semibold text-gray-900">System Health Monitor</h2>
        <button
          onClick={fetchHealth}
          disabled={loading}
          className="btn-secondary flex items-center gap-2"
        >
          <RefreshCw className={`w-4 h-4 ${loading ? 'animate-spin' : ''}`} />
          <span>Refresh</span>
        </button>
      </div>

      {/* Stats Grid */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {stats.map((stat, index) => {
          const Icon = stat.icon;
          return (
            <div key={index} className="card p-5">
              <div className="flex items-center gap-3">
                <div className={`p-3 rounded-lg ${stat.bgColor}`}>
                  <Icon className={`w-5 h-5 ${stat.color}`} />
                </div>
                <div className="flex-1">
                  <p className="text-sm text-gray-600 mb-1">{stat.label}</p>
                  <p className="text-2xl font-bold text-gray-900">
                    {typeof stat.value === 'string' ? (
                      <span className="capitalize">{stat.value}</span>
                    ) : (
                      stat.value
                    )}
                  </p>
                </div>
              </div>
            </div>
          );
        })}
      </div>

      {/* Cache Info */}
      <div className="card p-6">
        <h3 className="text-lg font-semibold text-gray-900 mb-4">Cache Information</h3>
        <div className="space-y-3">
          <div className="flex items-center justify-between py-2 border-b border-gray-100">
            <span className="text-sm text-gray-600">Cache Initialized</span>
            <span className="text-sm font-medium text-gray-900">
              {healthData.cache_initialized ? 'Yes' : 'No'}
            </span>
          </div>
          <div className="flex items-center justify-between py-2">
            <span className="text-sm text-gray-600">Last Updated</span>
            <span className="text-sm font-medium text-gray-900">
              {new Date().toLocaleString()}
            </span>
          </div>
        </div>
      </div>
    </div>
  );
};

export default HealthMonitor;
