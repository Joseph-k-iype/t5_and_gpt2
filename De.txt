import { useState } from 'react';
import { Search, AlertCircle } from 'lucide-react';
import LoadingSpinner from './LoadingSpinner';
import ResultDisplay from './ResultDisplay';
import api from '../services/api';

const EvaluationForm = () => {
  const [formData, setFormData] = useState({
    caseId: '',
    sendingCountry: '',
    receivingCountry: '',
    personalData: 'Yes',
  });
  const [result, setResult] = useState(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const handleChange = (field, value) => {
    setFormData((prev) => ({ ...prev, [field]: value }));
    setError(null);
  };

  const handleSubmit = async () => {
    // Validation
    if (!formData.sendingCountry || !formData.receivingCountry) {
      setError('Please fill in all required fields');
      return;
    }

    setLoading(true);
    setError(null);
    setResult(null);

    const { data, error: apiError } = await api.evaluateTransfer(formData);

    if (apiError) {
      setError(apiError);
    } else {
      setResult(data);
    }

    setLoading(false);
  };

  const handleReset = () => {
    setFormData({
      caseId: '',
      sendingCountry: '',
      receivingCountry: '',
      personalData: 'Yes',
    });
    setResult(null);
    setError(null);
  };

  return (
    <div className="space-y-6">
      {/* Form Card */}
      <div className="card p-6">
        <h2 className="text-xl font-semibold text-gray-900 mb-6">
          Transfer Evaluation Form
        </h2>

        <div className="space-y-4">
          {/* Row 1 */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div>
              <label htmlFor="caseId" className="label">
                Case ID <span className="text-gray-400">(Optional)</span>
              </label>
              <input
                id="caseId"
                type="text"
                className="input"
                placeholder="DVC123456, BA123, or 789"
                value={formData.caseId}
                onChange={(e) => handleChange('caseId', e.target.value)}
              />
              <p className="mt-1 text-xs text-gray-500">
                Supports DVC, Business Application, or numeric IDs
              </p>
            </div>

            <div>
              <label htmlFor="personalData" className="label">
                Personal Data
              </label>
              <select
                id="personalData"
                className="input"
                value={formData.personalData}
                onChange={(e) => handleChange('personalData', e.target.value)}
              >
                <option value="Yes">Yes</option>
                <option value="No">No</option>
              </select>
            </div>
          </div>

          {/* Row 2 */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div>
              <label htmlFor="sendingCountry" className="label">
                Sending Country <span className="text-red-500">*</span>
              </label>
              <input
                id="sendingCountry"
                type="text"
                className="input"
                placeholder="e.g., GB or United Kingdom"
                value={formData.sendingCountry}
                onChange={(e) => handleChange('sendingCountry', e.target.value)}
                required
              />
            </div>

            <div>
              <label htmlFor="receivingCountry" className="label">
                Receiving Country <span className="text-red-500">*</span>
              </label>
              <input
                id="receivingCountry"
                type="text"
                className="input"
                placeholder="e.g., US or United States"
                value={formData.receivingCountry}
                onChange={(e) => handleChange('receivingCountry', e.target.value)}
                required
              />
            </div>
          </div>

          {/* Error Message */}
          {error && (
            <div className="flex items-center gap-2 p-3 bg-red-50 border border-red-200 rounded-lg text-red-800 text-sm">
              <AlertCircle className="w-4 h-4 flex-shrink-0" />
              <span>{error}</span>
            </div>
          )}

          {/* Action Buttons */}
          <div className="flex items-center gap-3 pt-2">
            <button
              onClick={handleSubmit}
              disabled={loading}
              className="btn-primary flex items-center gap-2"
            >
              {loading ? (
                <>
                  <LoadingSpinner size="sm" />
                  <span>Evaluating...</span>
                </>
              ) : (
                <>
                  <Search className="w-4 h-4" />
                  <span>Evaluate Transfer</span>
                </>
              )}
            </button>
            <button
              onClick={handleReset}
              disabled={loading}
              className="btn-secondary"
            >
              Reset
            </button>
          </div>
        </div>
      </div>

      {/* Result Display */}
      {result && <ResultDisplay result={result} />}
    </div>
  );
};

export default EvaluationForm;
