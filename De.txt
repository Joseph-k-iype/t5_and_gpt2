let cur = 0;
const deck  = document.getElementById('deck');
const ctr   = document.getElementById('ctr');
const bar   = document.getElementById('progress');
const chartDone = {};

function build() {
  SLIDES.forEach((s,i) => {
    const d = document.createElement('div');
    d.className = 'slide'; d.id = `s${i}`;
    d.innerHTML = `
    <div class="s-header">
      <div class="s-header-accent"><span>${s.tag}</span></div>
      <div class="s-header-body">
        <span class="htitle">${s.title}</span>
        <span class="htag">GPU SIZING FRAMEWORK</span>
      </div>
    </div>
    ${s.html}`;
    deck.appendChild(d);
  });
}

function go(n) {
  const prev = document.querySelector('.slide.active');
  if (prev) { prev.classList.add('exiting'); prev.classList.remove('active'); setTimeout(()=>prev.classList.remove('exiting'),480); }
  cur = Math.max(0, Math.min(SLIDES.length-1, n));
  const next = document.getElementById(`s${cur}`);
  next.classList.add('active');
  ctr.textContent = `${cur+1} / ${SLIDES.length}`;
  bar.style.width = `${((cur+1)/SLIDES.length)*100}%`;
  drawChart(cur);
}

function drawChart(i) {
  if (chartDone[i]) return; chartDone[i] = true;
  if (i === 4) {
    const ctx = document.getElementById('chChart');
    if (!ctx) return;
    new Chart(ctx, {
      type:'bar',
      data:{
        labels:['1B','7B','13B','70B','405B','1T','6T (Opt)'],
        datasets:[{ label:'Tokens (T)', data:[0.02,0.14,0.26,1.4,8.1,20,120],
          backgroundColor:['#9DCFE0','#9DCFE0','#9DCFE0','#1B3D6F','#B5821A','#B5821A','#2A7A50'],
          borderRadius:4 }]
      },
      options:{
        responsive:true, maintainAspectRatio:false,
        plugins:{legend:{display:false}},
        scales:{
          x:{ticks:{color:'#7A8899',font:{size:10,family:'Outfit'}},grid:{display:false}},
          y:{ticks:{color:'#7A8899',font:{size:10}},grid:{color:'rgba(0,0,0,0.05)'},
             title:{display:true,text:'Tokens Required (Trillions)',color:'#7A8899',font:{size:10,family:'Outfit'}}}
        }
      }
    });
  }
}

document.getElementById('bn').onclick = ()=>go(cur+1);
document.getElementById('bp').onclick = ()=>go(cur-1);
document.addEventListener('keydown',e=>{
  if(e.key==='ArrowRight'||e.key==='ArrowDown'||e.key===' '){ e.preventDefault(); go(cur+1); }
  if(e.key==='ArrowLeft'||e.key==='ArrowUp'){ e.preventDefault(); go(cur-1); }
  if(e.key==='Home') go(0);
  if(e.key==='End')  go(SLIDES.length-1);
});

build(); go(0);
