    <script>
        // Update the updateMetricsCharts function to include PBT tagging jobs
        function updateMetricsCharts(data) {
            if (!data || data.length === 0) return;
            
            // Sort data by timestamp
            data.sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));
            
            // Format timestamps for display
            const timestamps = data.map(item => {
                const date = new Date(item.timestamp);
                return date.toLocaleTimeString([], {hour: '2-digit', minute: '2-digit'});
            });
            
            // Extract data points
            const cpuData = data.map(item => item.cpu_usage);
            const memoryData = data.map(item => item.memory_usage);
            const enhancementJobsData = data.map(item => item.enhancement_jobs_count);
            const taggingJobsData = data.map(item => item.tagging_jobs_count);
            const pbtTaggingJobsData = data.map(item => item.pbt_tagging_jobs_count || 0);
            
            // Update charts
            updateChart(charts.cpuChart, timestamps, cpuData);
            updateChart(charts.memoryChart, timestamps, memoryData);
            
            // Update jobs chart
            charts.jobsChart.data.labels = timestamps;
            charts.jobsChart.data.datasets[0].data = enhancementJobsData;
            charts.jobsChart.data.datasets[1].data = taggingJobsData;
            
            // Add PBT tagging jobs dataset if it doesn't exist
            if (charts.jobsChart.data.datasets.length < 3) {
                charts.jobsChart.data.datasets.push({
                    label: 'PBT Tagging Jobs',
                    data: pbtTaggingJobsData,
                    backgroundColor: 'rgba(139, 92, 246, 0.5)', // Purple color for PBT
                });
            } else {
                charts.jobsChart.data.datasets[2].data = pbtTaggingJobsData;
            }
            
            charts.jobsChart.update();
            
            // Update current metric values
            if (data.length > 0) {
                const latest = data[data.length - 1];
                document.getElementById('current-cpu').textContent = `${latest.cpu_usage.toFixed(1)}%`;
                document.getElementById('current-memory').textContent = `${latest.memory_usage.toFixed(1)}%`;
                document.getElementById('current-enhancement-jobs').textContent = latest.enhancement_jobs_count;
                document.getElementById('current-tagging-jobs').textContent = latest.tagging_jobs_count;
                document.getElementById('current-pbt-jobs').textContent = latest.pbt_tagging_jobs_count || 0;
            }
        }
        
        // Function to switch vector database type
        function switchVectorDB(type) {
            const containerElement = document.getElementById('vector-database-container');
            
            if (containerElement) {
                containerElement.innerHTML = `
                    <div class="flex items-center">
                        <div class="animate-spin mr-2">
                            <i class="fas fa-circle-notch"></i>
                        </div>
                        <span>Switching vector database to ${type}...</span>
                    </div>
                `;
            }
            
            fetch('/api/v1/settings/vector-database', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ type: type })
            })
            .then(response => {
                if (!response.ok) {
                    throw new Error(`HTTP error! Status: ${response.status}`);
                }
                return response.json();
            })
            .then(data => {
                showToast(`Vector database switched to ${type} successfully`, 'success');
                
                // Refresh components
                document.getElementById('vector-database-container').setAttribute('hx-trigger', 'load');
                htmx.process(document.getElementById('vector-database-container'));
                
                document.getElementById('vector-db-stats-container').setAttribute('hx-trigger', 'load');
                htmx.process(document.getElementById('vector-db-stats-container'));
            })
            .catch(error => {
                console.error('Error switching vector database:', error);
                showToast('Failed to switch vector database', 'error');
                
                // Refresh to show current state
                document.getElementById('vector-database-container').setAttribute('hx-trigger', 'load');
                htmx.process(document.getElementById('vector-database-container'));
            });
        }
    </script>
