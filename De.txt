PREFIX collibra: <http://your-collibra-namespace/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

CONSTRUCT {
    # All predicates and classes for Reports
    ?s ?p1 ?o1 .
    ?s rdf:type ?type1 .
    
    # All predicates and classes for Report Sections
    ?RS ?p2 ?o2 .
    ?RS rdf:type ?type2 .
    
    # All predicates and classes for Processes
    ?process ?p3 ?o3 .
    ?process rdf:type ?type3 .
    
    # All predicates and classes for Mappings
    ?map ?p4 ?o4 .
    ?map rdf:type ?type4 .
    
    # All predicates and classes for Consumed Data Elements
    ?consueddataelement ?p5 ?o5 .
    ?consueddataelement rdf:type ?type5 .
    
    # All predicates and classes for Consumed Apps
    ?consumedApp ?p6 ?o6 .
    ?consumedApp rdf:type ?type6 .
    
    # All predicates and classes for Produced Data Elements
    ?produceddataelement ?p7 ?o7 .
    ?produceddataelement rdf:type ?type7 .
    
    # All predicates and classes for Produced Apps
    ?producedApp ?p8 ?o8 .
    ?producedApp rdf:type ?type8 .
}
WHERE {
    {
        ?s a collibra:Report .
        ?RS collibra:isPartOf ?s .
        ?s collibra:associatedProcess ?process .
        ?map collibra:isAssociatedWithProcess ?process .
        ?map collibra:produces ?produceddataelement .
        ?map collibra:producedStoredInApp ?producedApp .
    }
    UNION
    {
        ?s a collibra:Report .
        ?RS collibra:isPartOf ?s .
        ?s collibra:associatedProcess ?process .
        ?map collibra:isAssociatedWithProcess ?process .
        ?map collibra:consumes ?consueddataelement .
        ?map collibra:consumedStoredInApp ?consumedApp .
    }
    UNION {
        ?s a collibra:Report .
        ?RS collibra:isPartOf ?s .
        ?process collibra:produces ?s .
        ?map collibra:isAssociatedWithProcess ?process .
        ?produceddataelement collibra:consumedBy ?RS .
        ?map collibra:produces ?produceddataelement .
        ?map collibra:producedStoredInApp ?producedApp .
    }
    
    # Get all predicates and objects for each subject
    OPTIONAL { ?s ?p1 ?o1 . }
    OPTIONAL { ?s rdf:type ?type1 . }
    
    OPTIONAL { ?RS ?p2 ?o2 . }
    OPTIONAL { ?RS rdf:type ?type2 . }
    
    OPTIONAL { ?process ?p3 ?o3 . }
    OPTIONAL { ?process rdf:type ?type3 . }
    
    OPTIONAL { ?map ?p4 ?o4 . }
    OPTIONAL { ?map rdf:type ?type4 . }
    
    OPTIONAL { ?consueddataelement ?p5 ?o5 . }
    OPTIONAL { ?consueddataelement rdf:type ?type5 . }
    
    OPTIONAL { ?consumedApp ?p6 ?o6 . }
    OPTIONAL { ?consumedApp rdf:type ?type6 . }
    
    OPTIONAL { ?produceddataelement ?p7 ?o7 . }
    OPTIONAL { ?produceddataelement rdf:type ?type7 . }
    
    OPTIONAL { ?producedApp ?p8 ?o8 . }
    OPTIONAL { ?producedApp rdf:type ?type8 . }
}
