# =============================================================================
# FalkorDB — Production-Grade Secure Docker Compose
# =============================================================================
# Image  : falkordb/falkordb:v4.16.3-alpine  (server + browser, one container)
#
# Credentials : read from .env file  (FALKORDB_USERNAME / FALKORDB_PASSWORD)
# TLS certs   : bind-mounted from ./secrets/{ca.crt, server.crt, server.key}
#
# Open ports  (ONLY these two are reachable from the host):
#   6379  — FalkorDB / Redis, TLS only.  Username + password required.
#   3000  — FalkorDB Browser UI.         Login requires username + password.
#
# Security model:
#   • Credentials come from .env — never hard-coded in this file
#   • Default Redis user disabled; only the named service account can connect
#   • TLS 1.2/1.3 on port 6379 — plain-text port disabled (port 0)
#   • Entrypoint writes a fresh ACL file from env vars on every container start
#   • ACL file lives in tmpfs (/tmp) — never written to disk
#   • Credentials cleared from env before Redis process starts
#   • Dangerous Redis commands renamed to "" (removed from command table)
#   • ACL blocks @dangerous and @admin for the service account
#   • Resource limits cap CPU and memory usage
#   • AOF + RDB persistence, fsync every second
#
# Prerequisites:
#   1. Create .env  (copy .env.example and fill in values)
#   2. Run  ./scripts/gen-tls-certs.sh  to populate secrets/
#   3. docker compose up -d
# =============================================================================

services:

  falkordb:
    image: falkordb/falkordb:v4.16.3-alpine
    container_name: falkordb-prod

    # -------------------------------------------------------------------------
    # PORT POLICY — exactly two ports, nothing else.
    # Lock these down at the host firewall to trusted source CIDRs:
    #   ufw allow from <trusted_cidr> to any port 6379
    #   ufw allow from <trusted_cidr> to any port 3000
    # -------------------------------------------------------------------------
    ports:
      - "6379:6379"   # FalkorDB / Redis — TLS only, username + password required
      - "3000:3000"   # Browser UI       — login requires username + password

    # -------------------------------------------------------------------------
    # Credentials — loaded from .env at compose-up time.
    # Required keys:  FALKORDB_USERNAME  and  FALKORDB_PASSWORD
    # See .env.example for the expected format.
    # -------------------------------------------------------------------------
    env_file:
      - .env

    # -------------------------------------------------------------------------
    # Non-sensitive tuning — safe to keep directly in this file.
    # -------------------------------------------------------------------------
    environment:
      FALKORDB_ARGS: >-
        THREAD_COUNT 8
        CACHE_SIZE 50
        TIMEOUT_MAX 60000
        TIMEOUT_DEFAULT 30000
        QUERY_MEM_CAPACITY 104857600

    # -------------------------------------------------------------------------
    # Entrypoint override
    #
    # What it does (in order):
    #   1. Validates FALKORDB_USERNAME and FALKORDB_PASSWORD are non-empty
    #   2. SHA-256 hashes the password — plaintext never written to disk
    #   3. Writes /tmp/users.acl into the tmpfs mount (memory-only)
    #   4. Sets REDIS_ARGS to load hardened redis.conf + the fresh ACL file
    #   5. Clears credentials from the environment
    #   6. Execs the real FalkorDB redis-server process (replaces this shell)
    # -------------------------------------------------------------------------
    entrypoint:
      - sh
      - -c
      - |
        set -euo pipefail

        # 1. Validate --------------------------------------------------------
        if [ -z "${FALKORDB_USERNAME:-}" ] || [ -z "${FALKORDB_PASSWORD:-}" ]; then
          echo "[falkordb] FATAL: FALKORDB_USERNAME and FALKORDB_PASSWORD must be set in .env"
          exit 1
        fi

        # 2. Hash the password -----------------------------------------------
        # Redis ACL format accepts #<sha256hex> so the plaintext never appears
        # in the ACL file or in any log output.
        PASS_HASH=$(echo -n "${FALKORDB_PASSWORD}" | sha256sum | awk '{print $1}')

        # 3. Write ACL file to tmpfs -----------------------------------------
        # /tmp is a tmpfs (see tmpfs: section below) — exists in memory only.
        cat > /tmp/users.acl << ACLEOF
        # FalkorDB ACL — generated at container start from .env.
        # Lives in tmpfs only; never written to the host filesystem.

        # Disable the built-in default user — no anonymous access whatsoever.
        user default off nopass nocommands nokeys

        # Service account — the ONLY user permitted to connect.
        #   +@all       allow all command categories …
        #   -@dangerous … except dangerous ones (FLUSHALL, DEBUG, OBJECT, etc.)
        #   -@admin     … and no admin commands (CONFIG, BGSAVE override, etc.)
        #   -CONFIG     belt-and-suspenders: CONFIG blocked even if @admin slips
        #   -DEBUG      belt-and-suspenders: DEBUG  blocked even if @admin slips
        #   -SHUTDOWN   prevent accidental shutdown from an app client
        #   -REPLICAOF  replication topology changes are an ops concern
        #   ~*          access all key patterns
        #   &*          access all pub/sub channels
        user ${FALKORDB_USERNAME} on #${PASS_HASH} ~* &* +@all -@dangerous -@admin -CONFIG -DEBUG -SHUTDOWN -REPLICAOF
        ACLEOF

        chmod 640 /tmp/users.acl

        # 4. Wire up REDIS_ARGS ----------------------------------------------
        # --include loads the hardened redis.conf (TLS, persistence, limits).
        # --aclfile points at the freshly written tmpfs ACL file.
        export REDIS_ARGS="--include /etc/falkordb/redis.conf --aclfile /tmp/users.acl"

        # 5. Clear credentials from the environment --------------------------
        # The ACL file is written; the plaintext is no longer needed.
        unset FALKORDB_USERNAME FALKORDB_PASSWORD PASS_HASH

        # 6. Exec the real FalkorDB entrypoint (this shell is replaced) ------
        exec docker-entrypoint.sh redis-server

    volumes:
      # Persistent graph data — RDB snapshots + AOF journal
      - falkordb_data:/data

      # Hardened redis.conf — TLS settings, disabled commands, persistence
      - ./config/redis.conf:/etc/falkordb/redis.conf:ro

      # TLS certificates — populated by ./scripts/gen-tls-certs.sh
      # server.key must be chmod 600 on the host before `docker compose up`
      - ./secrets/ca.crt:/etc/falkordb/tls/ca.crt:ro
      - ./secrets/server.crt:/etc/falkordb/tls/server.crt:ro
      - ./secrets/server.key:/etc/falkordb/tls/server.key:ro

    # ACL file is written here — tmpfs means it never touches the host disk
    tmpfs:
      - /tmp:mode=1770,size=16m

    restart: unless-stopped

    healthcheck:
      test:
        - "CMD-SHELL"
        - |
          redis-cli \
            --tls \
            --cacert /etc/falkordb/tls/ca.crt \
            -p 6379 \
            PING | grep -q PONG
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 40s

    networks:
      - falkordb-net

    deploy:
      resources:
        limits:
          cpus: "4.0"
          memory: 8G
        reservations:
          cpus: "1.0"
          memory: 2G

    security_opt:
      - no-new-privileges:true

    logging:
      driver: "json-file"
      options:
        max-size: "20m"
        max-file: "5"

# =============================================================================
# NETWORK — isolated bridge; only the two ports: mappings above reach the host
# =============================================================================
networks:
  falkordb-net:
    driver: bridge
    ipam:
      driver: default
      config:
        - subnet: 172.28.0.0/24
          gateway: 172.28.0.1
    driver_opts:
      com.docker.network.bridge.name: br-falkordb

# =============================================================================
# VOLUMES
# =============================================================================
volumes:
  falkordb_data:
    driver: local
    labels:
      com.falkordb.description: "FalkorDB persistent graph data (RDB + AOF)"
      com.falkordb.backup: "required"
