      ${s.html}`;
    deck.appendChild(div);
  });
}

function goTo(n) {
  const prev = document.querySelector('.slide.active');
  if (prev) { prev.classList.remove('active'); prev.classList.add('prev'); setTimeout(()=>prev.classList.remove('prev'),500); }
  current = Math.max(0, Math.min(slides.length-1, n));
  const next = document.getElementById(`slide-${current+1}`);
  next.classList.add('active');
  counter.textContent = `${current+1} / ${slides.length}`;
  bar.style.width = `${((current+1)/slides.length)*100}%`;
  drawCharts(current);
}

function drawCharts(idx) {
  if (chartDrawn[idx]) return;
  chartDrawn[idx] = true;
  if (idx === 4) { // Chinchilla chart
    const ctx = document.getElementById('chinchillaChart');
    if (ctx) new Chart(ctx, {
      type:'bar',
      data:{
        labels:['1B','7B','13B','70B','405B','1T','14.75T\n(Optimal)'],
        datasets:[{
          label:'Tokens Required (T)',
          data:[0.02,0.14,0.26,1.4,8.1,20,295],
          backgroundColor:['#0891B2','#0891B2','#0891B2','#00C8FF','#F59E0B','#7C3AED','#10B981'],
          borderRadius:4,
        }]
      },
      options:{
        responsive:true, maintainAspectRatio:false,
        plugins:{legend:{display:false},tooltip:{callbacks:{label:c=>`${c.raw}T tokens`}}},
        scales:{
          x:{ticks:{color:'#64748B',font:{size:10,family:'DM Sans'}},grid:{display:false}},
          y:{ticks:{color:'#64748B',font:{size:10}},grid:{color:'rgba(255,255,255,0.05)'},
             title:{display:true,text:'Tokens (Trillions)',color:'#64748B',font:{size:10}}}
        }
      }
    });
  }
}

document.getElementById('btn-next').onclick = ()=>goTo(current+1);
document.getElementById('btn-prev').onclick = ()=>goTo(current-1);
document.addEventListener('keydown',e=>{
  if(e.key==='ArrowRight'||e.key==='ArrowDown'||e.key===' ') { e.preventDefault(); goTo(current+1); }
  if(e.key==='ArrowLeft'||e.key==='ArrowUp')                  { e.preventDefault(); goTo(current-1); }
  if(e.key==='Home') goTo(0);
  if(e.key==='End')  goTo(slides.length-1);
});
