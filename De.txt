import React, { useState } from 'react';
import { CheckCircle, XCircle, AlertCircle, ChevronDown, ChevronRight } from 'lucide-react';

const ResultDisplay = ({ result }) => {
  const [expanded, setExpanded] = useState(false);

  const getStatusInfo = (status) => {
    if (!status) {
      return {
        icon: AlertCircle,
        bgColor: 'bg-gray-100',
        textColor: 'text-gray-700',
        borderColor: 'border-gray-200',
        iconColor: 'text-gray-600',
      };
    }

    if (status.includes('satisfied')) {
      return {
        icon: CheckCircle,
        bgColor: 'bg-green-50',
        textColor: 'text-green-800',
        borderColor: 'border-green-200',
        iconColor: 'text-green-600',
      };
    }

    if (status.includes('mismatch') || status.includes('not_satisfied')) {
      return {
        icon: XCircle,
        bgColor: 'bg-red-50',
        textColor: 'text-red-800',
        borderColor: 'border-red-200',
        iconColor: 'text-red-600',
      };
    }

    return {
      icon: AlertCircle,
      bgColor: 'bg-amber-50',
      textColor: 'text-amber-800',
      borderColor: 'border-amber-200',
      iconColor: 'text-amber-600',
    };
  };

  const statusInfo = getStatusInfo(result.status);
  const StatusIcon = statusInfo.icon;

  return (
    <div className="space-y-4">
      {/* Main Result Card */}
      <div className={`card p-6 ${statusInfo.borderColor} border-2`}>
        <div className="flex items-start gap-4">
          <div className={`p-3 rounded-lg ${statusInfo.bgColor}`}>
            <StatusIcon className={`w-6 h-6 ${statusInfo.iconColor}`} />
          </div>
          <div className="flex-1">
            <h3 className="text-lg font-semibold text-gray-900 mb-2">
              Evaluation Result
            </h3>
            
            {result.message && (
              <div className={`p-4 rounded-lg ${statusInfo.bgColor} mb-3`}>
                <p className={`font-medium ${statusInfo.textColor}`}>
                  {result.message}
                </p>
              </div>
            )}

            {result.status && (
              <div className="flex items-center gap-2 text-sm">
                <span className="text-gray-600">Status:</span>
                <span className={`font-semibold ${statusInfo.textColor}`}>
                  {result.status.replace(/_/g, ' ').toUpperCase()}
                </span>
              </div>
            )}
          </div>
        </div>
      </div>

      {/* Detailed Response Card */}
      <div className="card overflow-hidden">
        <button
          onClick={() => setExpanded(!expanded)}
          className="w-full px-6 py-4 flex items-center justify-between bg-gray-50 hover:bg-gray-100 transition-colors"
        >
          <span className="font-medium text-gray-900">Detailed Response</span>
          {expanded ? (
            <ChevronDown className="w-5 h-5 text-gray-500" />
          ) : (
            <ChevronRight className="w-5 h-5 text-gray-500" />
          )}
        </button>
        {expanded && (
          <div className="p-6 bg-white">
            <pre className="bg-gray-50 p-4 rounded-lg overflow-x-auto text-xs text-gray-800 border border-gray-200">
              {JSON.stringify(result, null, 2)}
            </pre>
          </div>
        )}
      </div>
    </div>
  );
};

export default ResultDisplay;
