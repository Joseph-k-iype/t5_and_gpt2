                <p style="font-size: 0.85rem; color: var(--hsbc-gray); margin-bottom: 1.5rem; font-weight: 300;">
                    Click on any row to view detailed case information
                </p>

                <div class="table-container" id="table-container">
                    <table id="results-table">
                        <thead>
                            <tr>
                                <th>Case ID â†“</th>
                                <th>EIM ID</th>
                                <th>Business App</th>
                                <th>Origin</th>
                                <th>Receiving</th>
                                <th>Purpose L1</th>
                                <th>Purpose L2</th>
                                <th>Purpose L3</th>

                                <th>TIA</th>
                                <th>HRPR</th>

                            </tr>

    <script>
        // API Base URL
        const API_BASE = '';

        // Initialize dashboard
        document.addEventListener('DOMContentLoaded', () => {
            loadStats();
            loadCountries();
            setupEventListeners();
        });

        // Load statistics
        async function loadStats() {
            try {
                const response = await fetch(`${API_BASE}/api/stats`);
                const data = await response.json();
                
                if (data.success) {
                    document.getElementById('stat-cases').textContent = 
                        data.stats.total_cases.toLocaleString();
                    document.getElementById('stat-countries').textContent = 
                        data.stats.total_countries.toLocaleString();
                    document.getElementById('stat-jurisdictions').textContent = 
                        data.stats.total_jurisdictions.toLocaleString();
                    document.getElementById('stat-pii').textContent = 
                        data.stats.cases_with_pii.toLocaleString();
                }
            } catch (error) {
                console.error('Error loading stats:', error);
            }
        }

        // Load countries
        async function loadCountries() {
            try {
                const response = await fetch(`${API_BASE}/api/countries`);
                const data = await response.json();
                
                if (data.success) {
                    const originList = document.getElementById('origin-countries-list');
                    const receivingList = document.getElementById('receiving-countries-list');
                    
                    data.countries.forEach(country => {
                        const option1 = document.createElement('option');
                        option1.value = country;
                        originList.appendChild(option1);
                        
                        const option2 = document.createElement('option');
                        option2.value = country;
                        receivingList.appendChild(option2);
                    });
                }
            } catch (error) {
                console.error('Error loading countries:', error);
            }
        }

        // Setup event listeners
        function setupEventListeners() {
            document.getElementById('search-form').addEventListener('submit', handleSearch);
            document.getElementById('reset-btn').addEventListener('click', resetForm);
            
            // Add real-time search listeners
            const searchFields = [
                'origin-country',
                'receiving-country', 
                'purpose-level1',
                'purpose-level2',
                'purpose-level3',
                'has-pii'
            ];
            
            searchFields.forEach(fieldId => {
                const field = document.getElementById(fieldId);
                field.addEventListener('change', handleRealtimeSearch);
                if (field.tagName === 'INPUT') {
                    field.addEventListener('input', debounce(handleRealtimeSearch, 500));
                }
            });
        }

        // Debounce function
        function debounce(func, wait) {
            let timeout;
            return function executedFunction(...args) {
                const later = () => {
                    clearTimeout(timeout);
                    func(...args);
                };
                clearTimeout(timeout);
                timeout = setTimeout(later, wait);
            };
        }

        // Handle real-time search
        async function handleRealtimeSearch() {
            const originCountry = document.getElementById('origin-country').value.trim();
            const receivingCountry = document.getElementById('receiving-country').value.trim();
            
            // Only search if we have both required fields
            if (!originCountry || !receivingCountry) {
                document.getElementById('search-summary').style.display = 'none';
                document.getElementById('rules-section').classList.remove('show');
                document.getElementById('results-section').classList.remove('show');
                return;
            }
            
            const purposeL1 = document.getElementById('purpose-level1').value;
            const purposeL2 = document.getElementById('purpose-level2').value;
            const purposeL3 = document.getElementById('purpose-level3').value;
            const piiValue = document.getElementById('has-pii').value; // 'yes', 'no', or ''
            
            // Update search summary
            updateSearchSummary(originCountry, receivingCountry, purposeL1, purposeL2, purposeL3, piiValue);
            
            // Perform search - send piiValue directly as string
            await performSearch(originCountry, receivingCountry, purposeL1, purposeL2, purposeL3, piiValue || null);
        }

        // Update search summary display
        function updateSearchSummary(origin, receiving, p1, p2, p3, pii) {
            document.getElementById('summary-origin').textContent = origin || '-';
            document.getElementById('summary-receiving').textContent = receiving || '-';
            document.getElementById('summary-purpose1').textContent = p1 || '-';
            document.getElementById('summary-purpose2').textContent = p2 || '-';
            document.getElementById('summary-purpose3').textContent = p3 || '-';
            document.getElementById('summary-pii').textContent = pii === 'yes' ? 'Yes' : pii === 'no' ? 'No' : '-';
            document.getElementById('search-summary').style.display = 'block';
        }

        // Handle search
        async function handleSearch(e) {
            e.preventDefault();
            
            const originCountry = document.getElementById('origin-country').value.trim();
            const receivingCountry = document.getElementById('receiving-country').value.trim();
            const purposeL1 = document.getElementById('purpose-level1').value;
            const purposeL2 = document.getElementById('purpose-level2').value;
            const purposeL3 = document.getElementById('purpose-level3').value;
            const piiValue = document.getElementById('has-pii').value; // 'yes', 'no', or ''
            
            if (!originCountry || !receivingCountry) {
                showAlert('Please enter both origin and receiving countries', 'error');
                return;
            }
            
            await performSearch(originCountry, receivingCountry, purposeL1, purposeL2, purposeL3, piiValue || null);
        }

        // Perform search (used by both manual search and real-time search)
        async function performSearch(originCountry, receivingCountry, purposeL1, purposeL2, purposeL3, hasPII) {
            // Show loading
            document.getElementById('loading').classList.add('show');
            document.getElementById('rules-section').classList.remove('show');
            document.getElementById('results-section').classList.remove('show');
            
            try {
                // Evaluate rules - convert hasPII for rules evaluation
                const hasPIIForRules = hasPII === true || hasPII === 'yes' ? true : 
                                       hasPII === false || hasPII === 'no' ? false : null;
                
                const rulesResponse = await fetch(`${API_BASE}/api/evaluate-rules`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        origin_country: originCountry,
                        receiving_country: receivingCountry,
                        has_pii: hasPIIForRules
                    })
                });
                
                const rulesData = await rulesResponse.json();
                
                if (rulesData.success) {
                    displayTriggeredRules(rulesData);
                    
                    // Search cases - send hasPII as string ('yes', 'no', or null)
                    const searchResponse = await fetch(`${API_BASE}/api/search-cases`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            origin_country: originCountry,
                            receiving_country: receivingCountry,
                            purpose_level1: purposeL1,
                            purpose_level2: purposeL2,
                            purpose_level3: purposeL3,
                            has_pii: hasPII, // Send as is (string 'yes'/'no' or null)
                            requirements: rulesData.requirements
                        })
                    });
                    
                    const searchData = await searchResponse.json();
                    
                    if (searchData.success) {
                        // Update summary count
                        document.getElementById('summary-count').textContent = searchData.total_cases.toLocaleString();
                        
                        displayResults(searchData.cases, rulesData.requirements);
                    } else {
                        showAlert(searchData.error, 'error');
                    }
                } else {
                    showAlert(rulesData.error, 'error');
                }
            } catch (error) {
                console.error('Search error:', error);
                showAlert('An error occurred while searching. Please try again.', 'error');
            } finally {
                document.getElementById('loading').classList.remove('show');
            }
        }

        // Display triggered rules
        function displayTriggeredRules(data) {
            const rulesSection = document.getElementById('rules-section');
            const rulesSummary = document.getElementById('rules-summary');
            const rulesList = document.getElementById('rules-list');
            
            const count = data.total_rules_triggered;
            rulesSummary.textContent = `${count} Compliance Rule${count !== 1 ? 's' : ''} Triggered`;
            
            rulesList.innerHTML = '';
            
            if (data.triggered_rules.length === 0) {
                rulesList.innerHTML = '<li class="rule-item">No specific compliance rules triggered for this transfer.</li>';
            } else {
                data.triggered_rules.forEach(rule => {
                    const li = document.createElement('li');
                    li.className = 'rule-item';
                    
                    const requirements = Object.entries(rule.requirements)
                        .map(([k, v]) => `${k.toUpperCase()}: ${v}`)
                        .join(', ');
                    
                    li.innerHTML = `
                        <div class="rule-id">${rule.rule_id}</div>
                        <div class="rule-desc">${rule.description}</div>
                        <div class="rule-requirements">Required Modules: ${requirements}</div>
                    `;
                    rulesList.appendChild(li);
                });
            }
            
            rulesSection.classList.add('show');
        }

        // Display results
        function displayResults(cases, requirements) {
            const resultsSection = document.getElementById('results-section');
            const resultsCount = document.getElementById('results-count');
            const resultsBody = document.getElementById('results-body');
            const noResults = document.getElementById('no-results');
            const tableContainer = document.getElementById('table-container');
            
            resultsCount.innerHTML = `Found <strong>${cases.length}</strong> matching case${cases.length !== 1 ? 's' : ''}`;
            
            if (cases.length === 0) {
                tableContainer.style.display = 'none';
                noResults.style.display = 'block';
            } else {
                tableContainer.style.display = 'block';
                noResults.style.display = 'none';
                
                resultsBody.innerHTML = '';
                
                cases.forEach((caseData, index) => {
                    const row = document.createElement('tr');
                    row.className = 'case-row';
                    row.dataset.caseIndex = index;
                    
                    const receivingCountries = Array.isArray(caseData.receiving_countries)
                        ? caseData.receiving_countries.join(', ')
                        : caseData.receiving_countries;
                    
                    // Check compliance
                    const complianceStatus = checkCompliance(caseData, requirements);
                    
                    row.innerHTML = `
                        <td><strong>${caseData.case_id || '-'}</strong></td>
                        <td>${caseData.eim_id || '-'}</td>
                        <td>${caseData.business_app_id || '-'}</td>
                        <td>${caseData.origin_country || '-'}</td>
                        <td>${receivingCountries || '-'}</td>
                        <td>${caseData.purpose_level1 || '-'}</td>
                        <td>${caseData.purpose_level2 || '-'}</td>
                        <td>${caseData.purpose_level3 || '-'}</td>
                        <td>${getModuleBadge(caseData.pia_module)}</td>
                        <td>${getModuleBadge(caseData.tia_module)}</td>
                        <td>${getModuleBadge(caseData.hrpr_module)}</td>
                        <td>${complianceStatus}</td>
                    `;
                    
                    // Add click handler to expand details
                    row.addEventListener('click', () => toggleCaseDetails(index, caseData, requirements));
                    
                    resultsBody.appendChild(row);
                    
                    // Create details row (hidden by default)
                    const detailsRow = document.createElement('tr');
                    detailsRow.className = 'case-details-row';
                    detailsRow.id = `details-${index}`;
                    detailsRow.innerHTML = `
                        <td colspan="12" class="case-details-cell">
                            <div class="case-details-content">
                                ${buildCaseDetailsHTML(caseData, requirements)}
                            </div>
                        </td>
                    `;
                    resultsBody.appendChild(detailsRow);
                });
            }
            
            resultsSection.classList.add('show');
        }

        // Toggle case details
        function toggleCaseDetails(index, caseData, requirements) {
            const detailsRow = document.getElementById(`details-${index}`);
            const caseRow = document.querySelector(`tr[data-case-index="${index}"]`);
            
            // Close all other details
            document.querySelectorAll('.case-details-row').forEach(row => {
                if (row.id !== `details-${index}`) {
                    row.classList.remove('show');
                }
            });
            document.querySelectorAll('.case-row').forEach(row => {
                if (row.dataset.caseIndex != index) {
                    row.classList.remove('expanded');
                }
            });
            
            // Toggle current details
            detailsRow.classList.toggle('show');
            caseRow.classList.toggle('expanded');
        }

        // Build case details HTML
        function buildCaseDetailsHTML(caseData, requirements) {
            const personalData = caseData.personal_data || [];
            const pdCategories = caseData.personal_data_categories || [];
            const categories = caseData.categories || [];
            
            return `
                <div class="detail-section">
                    <h4 class="detail-section-title">Case Information</h4>
                    <div class="detail-item"><strong>Case ID:</strong> ${caseData.case_id || '-'}</div>
                    <div class="detail-item"><strong>EIM ID:</strong> ${caseData.eim_id || '-'}</div>
                    <div class="detail-item"><strong>Business App ID:</strong> ${caseData.business_app_id || '-'}</div>
                    <div class="detail-item"><strong>Origin Country:</strong> ${caseData.origin_country || '-'}</div>
                    <div class="detail-item"><strong>Receiving Countries:</strong> ${caseData.receiving_countries.join(', ') || '-'}</div>
                </div>
                
                <div class="detail-section">
                    <h4 class="detail-section-title">Purpose Hierarchy</h4>
                    <div class="detail-item"><strong>Level 1:</strong> ${caseData.purpose_level1 || '-'}</div>
                    <div class="detail-item"><strong>Level 2:</strong> ${caseData.purpose_level2 || '-'}</div>
                    <div class="detail-item"><strong>Level 3:</strong> ${caseData.purpose_level3 || '-'}</div>
                </div>
                
                <div class="detail-section">
                    <h4 class="detail-section-title">Compliance Modules</h4>
                    <div class="detail-item"><strong>PIA Module:</strong> ${getModuleBadge(caseData.pia_module)}</div>
                    <div class="detail-item"><strong>TIA Module:</strong> ${getModuleBadge(caseData.tia_module)}</div>
                    <div class="detail-item"><strong>HRPR Module:</strong> ${getModuleBadge(caseData.hrpr_module)}</div>
                    <div class="detail-item" style="margin-top: 1rem; padding-top: 1rem; border-top: 1px solid var(--hsbc-border);">
                        <strong>Compliance Status:</strong><br>
                        ${checkCompliance(caseData, requirements)}
                    </div>
                </div>
                
                <div class="detail-section">
                    <h4 class="detail-section-title">Personal Data (${personalData.length})</h4>
                    ${personalData.length > 0 ? `
                        <ul class="detail-list">
                            ${personalData.map(pd => `<li>${pd}</li>`).join('')}
                        </ul>
                    ` : '<div class="detail-item">No personal data identified</div>'}
                </div>
                
                <div class="detail-section">
                    <h4 class="detail-section-title">Personal Data Categories (${pdCategories.length})</h4>
                    ${pdCategories.length > 0 ? `
                        <ul class="detail-list">
                            ${pdCategories.map(cat => `<li>${cat}</li>`).join('')}
                        </ul>
                    ` : '<div class="detail-item">No categories identified</div>'}
                </div>
                
                <div class="detail-section">
                    <h4 class="detail-section-title">Categories (${categories.length})</h4>
                    ${categories.length > 0 ? `
                        <ul class="detail-list">
                            ${categories.map(cat => `<li>${cat}</li>`).join('')}
                        </ul>
                    ` : '<div class="detail-item">No categories identified</div>'}
                </div>
            `;
        }

        // Check compliance against requirements
        function checkCompliance(caseData, requirements) {
            if (!requirements || Object.keys(requirements).length === 0) {
                return '<span class="compliance-badge compliance-pass">âœ“ No Rules</span>';
            }
            
            let missingModules = [];
            let hasModules = [];
            
            for (const [module, requiredValue] of Object.entries(requirements)) {
                const actualValue = caseData[module];
                
                if (actualValue === requiredValue) {
                    hasModules.push(module.replace('_module', '').toUpperCase());
                } else {
                    missingModules.push(module.replace('_module', '').toUpperCase());
                }
            }
            
            if (missingModules.length === 0) {
                return `<span class="compliance-badge compliance-pass">âœ“ Compliant</span>`;
            } else if (hasModules.length > 0) {
                return `<span class="compliance-badge compliance-partial">âš  Partial (Missing: ${missingModules.join(', ')})</span>`;
            } else {
                return `<span class="compliance-badge compliance-fail">âœ— Non-Compliant (Missing: ${missingModules.join(', ')})</span>`;
            }
        }

        // Get module badge
        function getModuleBadge(value) {
            if (!value || value === 'null' || value === 'None') {
                return '<span class="badge badge-warning">N/A</span>';
            }
            if (value === 'CM') {
                return '<span class="badge badge-success">CM</span>';
            }
            return `<span class="badge badge-info">${value}</span>`;
        }

        // Show alert
        function showAlert(message, type) {
            const container = document.getElementById('alert-container');
            const alert = document.createElement('div');
            alert.className = `alert alert-${type} show`;
            alert.textContent = message;
            container.appendChild(alert);
            
            setTimeout(() => {
                alert.classList.remove('show');
                setTimeout(() => alert.remove(), 300);
            }, 5000);
        }

        // Reset form
        function resetForm() {
            document.getElementById('search-form').reset();
            document.getElementById('search-summary').style.display = 'none';
            document.getElementById('rules-section').classList.remove('show');
            document.getElementById('results-section').classList.remove('show');
        }
    </script>
