    <script>
        // API Base URL
        const API_BASE = '';

        // Initialize dashboard
        document.addEventListener('DOMContentLoaded', () => {
            loadStats();
            loadCountries();
            setupEventListeners();
        });

        // Load statistics
        async function loadStats() {
            try {
                const response = await fetch(`${API_BASE}/api/stats`);
                const data = await response.json();
                
                if (data.success) {
                    document.getElementById('stat-cases').textContent = 
                        data.stats.total_cases.toLocaleString();
                    document.getElementById('stat-countries').textContent = 
                        data.stats.total_countries.toLocaleString();
                    document.getElementById('stat-jurisdictions').textContent = 
                        data.stats.total_jurisdictions.toLocaleString();
                    document.getElementById('stat-pii').textContent = 
                        data.stats.cases_with_pii.toLocaleString();
                }
            } catch (error) {
                console.error('Error loading stats:', error);
            }
        }

        // Load countries
        async function loadCountries() {
            try {
                const response = await fetch(`${API_BASE}/api/countries`);
                const data = await response.json();
                
                if (data.success) {
                    const originList = document.getElementById('origin-countries-list');
                    const receivingList = document.getElementById('receiving-countries-list');
                    
                    data.countries.forEach(country => {
                        const option1 = document.createElement('option');
                        option1.value = country;
                        originList.appendChild(option1);
                        
                        const option2 = document.createElement('option');
                        option2.value = country;
                        receivingList.appendChild(option2);
                    });
                }
            } catch (error) {
                console.error('Error loading countries:', error);
            }
        }

        // Setup event listeners
        function setupEventListeners() {
            document.getElementById('search-form').addEventListener('submit', handleSearch);
            document.getElementById('reset-btn').addEventListener('click', resetForm);
        }

        // Handle search
        async function handleSearch(e) {
            e.preventDefault();
            
            const originCountry = document.getElementById('origin-country').value.trim();
            const receivingCountry = document.getElementById('receiving-country').value.trim();
            const purposeL1 = document.getElementById('purpose-level1').value;
            const purposeL2 = document.getElementById('purpose-level2').value;
            const purposeL3 = document.getElementById('purpose-level3').value;
            const hasPII = document.getElementById('has-pii').checked;
            
            if (!originCountry || !receivingCountry) {
                showAlert('Please enter both origin and receiving countries', 'error');
                return;
            }
            
            // Show loading
            document.getElementById('loading').classList.add('show');
            document.getElementById('rules-section').classList.remove('show');
            document.getElementById('results-section').classList.remove('show');
            
            try {
                // Evaluate rules
                const rulesResponse = await fetch(`${API_BASE}/api/evaluate-rules`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        origin_country: originCountry,
                        receiving_country: receivingCountry,
                        has_pii: hasPII
                    })
                });
                
                const rulesData = await rulesResponse.json();
                
                if (rulesData.success) {
                    displayTriggeredRules(rulesData);
                    
                    // Search cases
                    const searchResponse = await fetch(`${API_BASE}/api/search-cases`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            origin_country: originCountry,
                            receiving_country: receivingCountry,
                            purpose_level1: purposeL1,
                            purpose_level2: purposeL2,
                            purpose_level3: purposeL3,
                            has_pii: hasPII,
                            requirements: rulesData.requirements
                        })
                    });
                    
                    const searchData = await searchResponse.json();
                    
                    if (searchData.success) {
                        displayResults(searchData.cases);
                    } else {
                        showAlert(searchData.error, 'error');
                    }
                } else {
                    showAlert(rulesData.error, 'error');
                }
            } catch (error) {
                console.error('Search error:', error);
                showAlert('An error occurred while searching. Please try again.', 'error');
            } finally {
                document.getElementById('loading').classList.remove('show');
            }
        }

        // Display triggered rules
        function displayTriggeredRules(data) {
            const rulesSection = document.getElementById('rules-section');
            const rulesSummary = document.getElementById('rules-summary');
            const rulesList = document.getElementById('rules-list');
            
            const count = data.total_rules_triggered;
            rulesSummary.textContent = `${count} Compliance Rule${count !== 1 ? 's' : ''} Triggered`;
            
            rulesList.innerHTML = '';
            
            if (data.triggered_rules.length === 0) {
                rulesList.innerHTML = '<li class="rule-item">No specific compliance rules triggered for this transfer.</li>';
            } else {
                data.triggered_rules.forEach(rule => {
                    const li = document.createElement('li');
                    li.className = 'rule-item';
                    
                    const requirements = Object.entries(rule.requirements)
                        .map(([k, v]) => `${k.toUpperCase()}: ${v}`)
                        .join(', ');
                    
                    li.innerHTML = `
                        <div class="rule-id">${rule.rule_id}</div>
                        <div class="rule-desc">${rule.description}</div>
                        <div class="rule-requirements">Required Modules: ${requirements}</div>
                    `;
                    rulesList.appendChild(li);
                });
            }
            
            rulesSection.classList.add('show');
        }

        // Display results
        function displayResults(cases) {
            const resultsSection = document.getElementById('results-section');
            const resultsCount = document.getElementById('results-count');
            const resultsBody = document.getElementById('results-body');
            const noResults = document.getElementById('no-results');
            const tableContainer = document.getElementById('table-container');
            
            resultsCount.innerHTML = `Found <strong>${cases.length}</strong> matching case${cases.length !== 1 ? 's' : ''}`;
            
            if (cases.length === 0) {
                tableContainer.style.display = 'none';
                noResults.style.display = 'block';
            } else {
                tableContainer.style.display = 'block';
                noResults.style.display = 'none';
                
                resultsBody.innerHTML = '';
                
                cases.forEach(caseData => {
                    const row = document.createElement('tr');
                    
                    const receivingCountries = Array.isArray(caseData.receiving_countries)
                        ? caseData.receiving_countries.join(', ')
                        : caseData.receiving_countries;
                    
                    row.innerHTML = `
                        <td><strong>${caseData.case_id || '-'}</strong></td>
                        <td>${caseData.eim_id || '-'}</td>
                        <td>${caseData.business_app_id || '-'}</td>
                        <td>${caseData.origin_country || '-'}</td>
                        <td>${receivingCountries || '-'}</td>
                        <td>${caseData.purpose_level1 || '-'}</td>
                        <td>${caseData.purpose_level2 || '-'}</td>
                        <td>${caseData.purpose_level3 || '-'}</td>
                        <td>${getModuleBadge(caseData.pia_module)}</td>
                        <td>${getModuleBadge(caseData.tia_module)}</td>
                        <td>${getModuleBadge(caseData.hrpr_module)}</td>
                    `;
                    
                    resultsBody.appendChild(row);
                });
            }
            
            resultsSection.classList.add('show');
        }

        // Get module badge
        function getModuleBadge(value) {
            if (!value || value === 'null' || value === 'None') {
                return '<span class="badge badge-warning">N/A</span>';
            }
            if (value === 'CM') {
                return '<span class="badge badge-success">CM</span>';
            }
            return `<span class="badge badge-info">${value}</span>`;
        }

        // Show alert
        function showAlert(message, type) {
            const container = document.getElementById('alert-container');
            const alert = document.createElement('div');
            alert.className = `alert alert-${type} show`;
            alert.textContent = message;
            container.appendChild(alert);
            
            setTimeout(() => {
                alert.classList.remove('show');
                setTimeout(() => alert.remove(), 300);
            }, 5000);
        }

        // Reset form
        function resetForm() {
            document.getElementById('search-form').reset();
            document.getElementById('rules-section').classList.remove('show');
            document.getElementById('results-section').classList.remove('show');
        }
